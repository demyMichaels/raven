3.7 Containers

Much computing involves creating collections of various forms of objects and then manipulating such collections. Reading characters into a string and printing out the string is a simple example. A class with the main purpose of holding objects is commonly called a container.

Why Containers Matter

Providing suitable containers for a given task and supporting them with useful fundamental operations are important steps in the construction of any program.

Real-World Application

To illustrate the standard library's most useful containers, consider a simple program for keeping names and telephone numbers. This is the kind of program for which different approaches appear "simple and obvious" to people of different backgrounds.

3.7.1 Vector Tutorial

The Old Way: Built-in Arrays

For many C programmers, a built-in array would seem to be a suitable starting point:

struct Entry {
    string name;
    int number;
};

Entry phone_book[1000];

void print_entry(int i) // simple use
{
    cout << phone_book[i].name << ' ' << phone_book[i].number << '\n';
}

The Problem:

A built-in array has a fixed size. If we choose a large size, we waste space; if we choose a smaller size, the array will overflow. In either case, we will have to write low-level memory-management code.

The Modern Way: Vectors

The standard library provides a vector that takes care of memory management:

vector<Entry> phone_book(1000);

void print_entry(int i) // simple use, exactly as for array
{
    cout << phone_book[i].name << ' ' << phone_book[i].number << '\n';
}

void add_entries(int n) // increase size by n
{
    phone_book.resize(phone_book.size() + n);
}

Key Points:

  • The size() member function returns the number of elements
  • Vectors can be dynamically resized with resize()
  • Syntax is very similar to arrays for basic operations

Important: Parentheses vs Brackets

Correct ‚úì

// Vector of 1000 elements
vector<Entry> book(1000);

Wrong ‚úó

// 1000 empty vectors!
vector<Entry> books[1000];

Note the use of parentheses () in the definition. We made a single object of type vector<Entry> and supplied its initial size as an initializer.

Vector Assignment and Copying

A vector is a single object that can be assigned:

void f(vector<Entry>& v)
{
    vector<Entry> v2 = phone_book;
    v = v2;
    // ...
}

‚ö†Ô∏è Performance Warning

Assigning a vector involves copying its elements. After the initialization and assignment in f(), v and v2 each holds a separate copy of every Entry in the phone book. When a vector holds many elements, such innocent-looking assignments can be prohibitively expensive.

Solution: Where copying is undesirable, references or pointers should be used.

üìö Student Management System

Managing dynamic student rosters with automatic resizing as students enroll or drop courses. 

struct Student {
    string name;
    string email;
    int student_id;
    vector<double> grades;
    double gpa;
};

class CourseRoster {
private:
    vector<Student> students;
    string course_name;
    int max_capacity;

public:
    CourseRoster(string name, int capacity)
        : course_name(name), max_capacity(capacity) {
        students.reserve(capacity); // Pre-allocate memory
    }

    bool enroll_student(const Student& s) {
        if (students.size() < max_capacity) {
            students.push_back(s);
            return true;
        }
        return false; // Course full
    }

    void drop_student(int student_id) {
        students.erase(
            remove_if(students.begin(), students.end(),
                [student_id](const Student& s) {
                    return s.student_id == student_id;
                }),
            students.end()
        );
    }

    double calculate_class_average() {
        if (students.empty()) return 0.0;

        double sum = 0.0;
        for (const auto& student : students) {
            sum += student.gpa;
        }
        return sum / students.size();
    }

    vector<Student> get_honor_students(double threshold = 3.5) {
        vector<Student> honors;
        for (const auto& student : students) {
            if (student.gpa >= threshold) {
                honors.push_back(student);
            }
        }
        return honors;
    }
};

struct Question {
    string text;
    vector<string> options;
    int correct_answer;
    string difficulty; // "easy", "medium", "hard"
    vector<string> tags; // topics covered
};

class AdaptiveQuiz {
private:
    vector<Question> question_bank;
    vector<Question> active_questions;
    vector<bool> answers;
    double current_performance;

public:
    void load_questions(const vector<Question>& questions) {
        question_bank = questions;
    }

    void generate_adaptive_quiz(int num_questions) {
        active_questions.clear();

        // Start with medium difficulty
        string target_difficulty = "medium";

        for (int i = 0; i < num_questions; i++) {
            // Select question based on performance
            Question q = select_question_by_difficulty(target_difficulty);
            active_questions.push_back(q);

            // Adjust difficulty based on answers
            if (i > 0 && current_performance > 0.8) {
                target_difficulty = "hard";
            } else if (i > 0 && current_performance < 0.5) {
                target_difficulty = "easy";
            }
        }
    }

    void submit_answer(int question_idx, int answer) {
        bool correct = (active_questions[question_idx].correct_answer == answer);
        answers.push_back(correct);
        update_performance();
    }

    void update_performance() {
        int correct_count = 0;
        for (bool ans : answers) {
            if (ans) correct_count++;
        }
        current_performance = static_cast<double>(correct_count) / answers.size();
    }

    vector<Question> get_questions_by_tag(const string& tag) {
        vector<Question> filtered;
        for (const auto& q : question_bank) {
            for (const auto& t : q.tags) {
                if (t == tag) {
                    filtered.push_back(q);
                    break;
                }
            }
        }
        return filtered;
    }

private:
    Question select_question_by_difficulty(const string& diff) {
        vector<Question> candidates;
        for (const auto& q : question_bank) {
            if (q.difficulty == diff) {
                candidates.push_back(q);
            }
        }
        // Return random question from candidates
        return candidates[rand() % candidates.size()];
    }
};

üíº Portfolio Management System

Dynamic portfolio tracking with automatic rebalancing and risk assessment. 

struct Asset {
    string symbol;
    string asset_type; // "stock", "bond", "etf", "crypto"
    double quantity;
    double purchase_price;
    double current_price;
    string purchase_date;
};

class Portfolio {
private:
    string client_id;
    vector<Asset> holdings;
    double cash_balance;

public:
    Portfolio(string id, double initial_cash)
        : client_id(id), cash_balance(initial_cash) {}

    bool buy_asset(const Asset& asset) {
        double cost = asset.quantity * asset.purchase_price;
        if (cash_balance >= cost) {
            holdings.push_back(asset);
            cash_balance -= cost;
            return true;
        }
        return false; // Insufficient funds
    }

    void sell_asset(const string& symbol, double quantity) {
        for (auto it = holdings.begin(); it != holdings.end(); ++it) {
            if (it->symbol == symbol) {
                double proceeds = quantity * it->current_price;
                cash_balance += proceeds;

                it->quantity -= quantity;
                if (it->quantity <= 0) {
                    holdings.erase(it);
                }
                break;
            }
        }
    }

    double get_total_value() {
        double total = cash_balance;
        for (const auto& asset : holdings) {
            total += asset.quantity * asset.current_price;
        }
        return total;
    }

    double get_total_return() {
        double total_invested = 0.0;
        double current_value = 0.0;

        for (const auto& asset : holdings) {
            total_invested += asset.quantity * asset.purchase_price;
            current_value += asset.quantity * asset.current_price;
        }

        return ((current_value - total_invested) / total_invested) * 100.0;
    }

    map<string, double> get_asset_allocation() {
        map<string, double> allocation;
        double total = get_total_value();

        for (const auto& asset : holdings) {
            double value = asset.quantity * asset.current_price;
            allocation[asset.asset_type] += (value / total) * 100.0;
        }

        return allocation;
    }

    vector<Asset> get_top_performers(int count = 5) {
        vector<Asset> performers = holdings;

        // Sort by return percentage
        sort(performers.begin(), performers.end(),
            [](const Asset& a, const Asset& b) {
                double return_a = ((a.current_price - a.purchase_price) /
                                   a.purchase_price) * 100.0;
                double return_b = ((b.current_price - b.purchase_price) /
                                   b.purchase_price) * 100.0;
                return return_a > return_b;
            });

        if (performers.size() > count) {
            performers.resize(count);
        }

        return performers;
    }

    bool needs_rebalancing(const map<string, double>& target_allocation,
                          double tolerance = 5.0) {
        auto current = get_asset_allocation();

        for (const auto& [asset_type, target_pct] : target_allocation) {
            double current_pct = current[asset_type];
            if (abs(current_pct - target_pct) > tolerance) {
                return true;
            }
        }
        return false;
    }
};

        for (const auto& txn : transactions) {

                // Extract month from timestamp (simplified)
                string month = txn.timestamp.substr(0, 7); // YYYY-MM
                monthly_totals[month] += txn.amount;
            }

        if (monthly_totals.empty()) return 0.0;

        double total = 0.0;
        for (const auto& [month, amount] : monthly_totals) {
            total += amount;
        }

                merchant_frequency[txn.merchant]++;

        vector<string> recurring;

            if (freq >= 3) { // Appears 3+ times
                recurring.push_back(merchant);
            }
        }

        return recurring;
    }
};

üìà Market Data & Trading Analytics

Real-time market data processing and technical analysis. 

struct PriceData {
    string symbol;
    double open;
    double high;
    double low;
    double close;
    long volume;
    string timestamp;
};

class MarketAnalyzer {
private:
    vector<PriceData> historical_data;

public:
    void load_historical_data(const vector<PriceData>& data) {
        historical_data = data;
        // Sort by timestamp
        sort(historical_data.begin(), historical_data.end(),
            [](const PriceData& a, const PriceData& b) {
                return a.timestamp < b.timestamp;
            });
    }

    vector<double> calculate_sma(int period) {
        // Simple Moving Average
        vector<double> sma;

        if (historical_data.size() < period) return sma;

        for (size_t i = period - 1; i < historical_data.size(); i++) {
            double sum = 0.0;
            for (int j = 0; j < period; j++) {
                sum += historical_data[i - j].close;
            }
            sma.push_back(sum / period);
        }

        return sma;
    }

    vector<double> calculate_ema(int period) {
        // Exponential Moving Average
        vector<double> ema;
        if (historical_data.empty()) return ema;

        double multiplier = 2.0 / (period + 1);
        double ema_prev = historical_data[0].close;
        ema.push_back(ema_prev);

        for (size_t i = 1; i < historical_data.size(); i++) {
            double ema_current = (historical_data[i].close - ema_prev) *
                                 multiplier + ema_prev;
            ema.push_back(ema_current);
            ema_prev = ema_current;
        }

        return ema;
    }

    double calculate_volatility(int period = 20) {
        if (historical_data.size() < period) return 0.0;

        vector<double> returns;
        for (size_t i = 1; i < min(historical_data.size(),
                                    static_cast<size_t>(period)); i++) {
            double ret = (historical_data[i].close -
                         historical_data[i-1].close) /
                         historical_data[i-1].close;
            returns.push_back(ret);
        }

        // Calculate standard deviation
        double mean = 0.0;
        for (double r : returns) mean += r;
        mean /= returns.size();

        double variance = 0.0;
        for (double r : returns) {
            variance += (r - mean) * (r - mean);
        }
        variance /= returns.size();

        return sqrt(variance);
    }

    struct SupportResistance {
        vector<double> support_levels;
        vector<double> resistance_levels;
    };

    SupportResistance find_support_resistance() {
        SupportResistance levels;

        // Find local minima (support) and maxima (resistance)
        for (size_t i = 1; i < historical_data.size() - 1; i++) {
            double prev = historical_data[i-1].close;
            double curr = historical_data[i].close;
            double next = historical_data[i+1].close;

            if (curr < prev && curr < next) {
                levels.support_levels.push_back(curr);
            }
            if (curr > prev && curr > next) {
                levels.resistance_levels.push_back(curr);
            }
        }

        return levels;
    }

    string generate_signal() {
        if (historical_data.size() < 50) return "HOLD - Insufficient data";

        auto sma_20 = calculate_sma(20);
        auto sma_50 = calculate_sma(50);

        if (sma_20.empty() || sma_50.empty()) return "HOLD";

        double current_sma_20 = sma_20.back();
        double current_sma_50 = sma_50.back();

        if (current_sma_20 > current_sma_50 * 1.02) {
            return "BUY - Bullish crossover detected";
        } else if (current_sma_20 < current_sma_50 * 0.98) {
            return "SELL - Bearish crossover detected";
        }

        return "HOLD - No clear signal";
    }
};
