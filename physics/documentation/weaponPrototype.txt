    <script>
        // Game Constants
        const CANVAS_WIDTH = window.innerWidth;
        const CANVAS_HEIGHT = window.innerHeight;
        const PLAYER_SPEED = 5;
        const BULLET_SPEED = 10;
        
        // Weapon Configurations
        const WEAPONS = {
            anvil: {
                name: "THE ANVIL",
                mass: 13.0,
                movementModifier: 0.7,
                bulletSpeed: 850,
                fireRate: 100, // ms between shots
                recoilPattern: 'climbing',
                ammoType: "7.62 NATO",
                maxAmmo: 100,
                clipSize: 50,
                heatPerShot: 3.5,
                maxHeat: 100,
                cooldownRate: 15, // per second
                recoilX: 0.3,
                recoilY: 0.8,
                climbRate: 0.12
            },
            judge: {
                name: "THE JUDGE",
                mass: 1.2,
                movementModifier: 1.05,
                bulletSpeed: 260,
                fireRate: 250,
                recoilPattern: 'snap',
                ammoType: ".45 ACP",
                maxAmmo: 56,
                clipSize: 8,
                recoilX: 0.1,
                recoilY: 1.5,
                recoverySpeed: 2.5
            },
            sweeper: {
                name: "ROOM SWEEPER",
                mass: 3.8,
                movementModifier: 0.9,
                bulletSpeed: 400,
                fireRate: 800,
                recoilPattern: 'impulse',
                ammoType: "12 Gauge",
                maxAmmo: 28,
                clipSize: 7,
                recoilX: 0.5,
                recoilY: 2.2,
                ammoMode: 'buckshot' // 'buckshot' or 'slug'
            }
        };

        // Game State
        let currentWeapon = 'judge';
        let player = {
            x: CANVAS_WIDTH / 2,
            y: CANVAS_HEIGHT / 2,
            angle: 0,
            velocity: { x: 0, y: 0 }
        };
        let bullets = [];
        let recoil = { x: 0, y: 0 };
        let heat = 0;
        let ammo = WEAPONS.judge.clipSize;
        let maxAmmo = WEAPONS.judge.maxAmmo;
        let lastFireTime = 0;
        let isOverheated = false;
        let isRecovering = false;
        let shells = WEAPONS.sweeper.clipSize;
        let target = { x: CANVAS_WIDTH / 2 + 200, y: CANVAS_HEIGHT / 2 };
        let keys = {};
        let cameraShake = 0;
        let screenShakeOffset = { x: 0, y: 0 };

        // DOM Elements
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const heatFill = document.getElementById('heatFill');
        const ammoCount = document.getElementById('ammoCount');
        const ammoType = document.getElementById('ammoType');
        const fpsCounter = document.getElementById('fpsCounter');

        // Initialize Canvas
        canvas.width = CANVAS_WIDTH;
        canvas.height = CANVAS_HEIGHT;

        // Input Handling
        document.addEventListener('keydown', (e) => {
            keys[e.key.toLowerCase()] = true;
            
            // Weapon switching
            if (e.key === '1') selectWeapon('anvil');
            if (e.key === '2') selectWeapon('judge');
            if (e.key === '3') selectWeapon('sweeper');
            if (e.key === 'Tab') toggleAmmoMode();
            if (e.key === 'r') reload();
        });

        document.addEventListener('keyup', (e) => {
            keys[e.key.toLowerCase()] = false;
        });

        canvas.addEventListener('mousedown', (e) => {
            if (e.button === 0) { // Left click
                fire();
            }
        });

        canvas.addEventListener('mousemove', (e) => {
            const rect = canvas.getBoundingClientRect();
            target.x = e.clientX - rect.left;
            target.y = e.clientY - rect.top;
        });

        // Weapon Functions
        function selectWeapon(weapon) {
            document.querySelectorAll('.weapon-card').forEach(card => {
                card.classList.remove('active');
            });
            document.getElementById(`weapon-${weapon}`).classList.add('active');
            
            currentWeapon = weapon;
            updateAmmoDisplay();
            
            // Reset weapon-specific states
            if (weapon === 'anvil') {
                heatFill.style.background = 'linear-gradient(90deg, #ff6b00, #ff0000)';
                document.getElementById('heatGauge').style.display = 'block';
            } else {
                document.getElementById('heatGauge').style.display = 'none';
            }
            
            if (weapon === 'sweeper') {
                ammoType.textContent = WEAPONS.sweeper.ammoMode === 'buckshot' ? 'Buckshot' : 'Slug';
            }
        }

        function toggleAmmoMode() {
            if (currentWeapon === 'sweeper') {
                WEAPONS.sweeper.ammoMode = WEAPONS.sweeper.ammoMode === 'buckshot' ? 'slug' : 'buckshot';
                ammoType.textContent = WEAPONS.sweeper.ammoMode === 'buckshot' ? 'Buckshot' : 'Slug';
                document.getElementById('sweeper-ammo').textContent = 
                    WEAPONS.sweeper.ammoMode === 'buckshot' ? 'Buckshot' : 'Slug';
            }
        }

        function fire() {
            const now = Date.now();
            const weapon = WEAPONS[currentWeapon];
            
            // Check cooldowns
            if (now - lastFireTime < weapon.fireRate) return;
            if (isOverheated) return;
            if (isRecovering) return;
            
            // Check ammo
            if ((currentWeapon === 'anvil' || currentWeapon === 'judge') && ammo <= 0) {
                reload();
                return;
            }
            if (currentWeapon === 'sweeper' && shells <= 0) {
                reload();
                return;
            }
            
            lastFireTime = now;
            
            // Weapon-specific firing logic
            switch(currentWeapon) {
                case 'anvil':
                    fireAnvil();
                    break;
                case 'judge':
                    fireJudge();
                    break;
                case 'sweeper':
                    fireSweeper();
                    break;
            }
            
            // Apply recoil
            applyRecoil();
            
            // Update UI
            updateAmmoDisplay();
            
            // Play fire effects
            playHeavyFireEffects();
            applyScreenShake(0.3, 0.15);
        }

        function fireAnvil() {
            // Heat management
            heat += WEAPONS.anvil.heatPerShot;
            if (heat >= WEAPONS.anvil.maxHeat) {
                isOverheated = true;
                setTimeout(() => {
                    isOverheated = false;
                    heat = 0;
                }, 2000);
            }
            
            // Update heat bar
            heatFill.style.width = `${(heat / WEAPONS.anvil.maxHeat) * 100}%`;
            
            // Create bullet
            ammo--;
            createBullet();
        }

        function fireJudge() {
            ammo--;
            isRecovering = true;
            
            // Calculate leading for moving target (simplified)
            const dx = target.x - player.x;
            const dy = target.y - player.y;
            const distance = Math.sqrt(dx * dx + dy * dy);
            const timeToTarget = distance / (WEAPONS.judge.bulletSpeed / 10);
            
            // Simulate bullet drop with slower velocity
            createBullet(WEAPONS.judge.bulletSpeed / 85);
            
            // Fast recovery
            setTimeout(() => {
                isRecovering = false;
            }, WEAPONS.judge.fireRate / 2);
        }

        function fireSweeper() {
            shells--;
            
            if (WEAPONS.sweeper.ammoMode === 'buckshot') {
                // Fire multiple pellets
                for (let i = 0; i < 8; i++) {
                    const spread = Math.random() * 0.2 - 0.1;
                    createBullet(WEAPONS.sweeper.bulletSpeed / 40, spread);
                }
            } else {
                // Fire single slug
                createBullet(WEAPONS.sweeper.bulletSpeed / 40);
            }
            
            // Pump action delay
            isRecovering = true;
            setTimeout(() => {
                isRecovering = false;
            }, WEAPONS.sweeper.fireRate);
            
            // Update shells bar
            document.getElementById('sweeper-shells-bar').style.width = 
                `${(shells / WEAPONS.sweeper.clipSize) * 100}%`;
        }

        function createBullet(speedMultiplier = 1, spread = 0) {
            const dx = target.x - player.x + recoil.x * 50;
            const dy = target.y - player.y + recoil.y * 50;
            const angle = Math.atan2(dy, dx) + spread;
            
            bullets.push({
                x: player.x,
                y: player.y,
                vx: Math.cos(angle) * BULLET_SPEED * speedMultiplier,
                vy: Math.sin(angle) * BULLET_SPEED * speedMultiplier,
                type: currentWeapon,
                life: 100
            });
        }

        function applyRecoil() {
            const weapon = WEAPONS[currentWeapon];
            
            switch(weapon.recoilPattern) {
                case 'climbing':
                    recoil.x += weapon.recoilX * (Math.random() * 0.4 - 0.2);
                    recoil.y += weapon.recoilY;
                    break;
                case 'snap':
                    recoil.x = weapon.recoilX * (Math.random() * 2 - 1);
                    recoil.y = weapon.recoilY * (0.8 + Math.random() * 0.4);
                    break;
                case 'impulse':
                    recoil.x = weapon.recoilX * (Math.random() * 2 - 1);
                    recoil.y = weapon.recoilY * (0.9 + Math.random() * 0.2);
                    break;
            }
            
            // Add recoil indicator
            createRecoilIndicator();
        }

        function reload() {
            const weapon = WEAPONS[currentWeapon];
            
            if (currentWeapon === 'sweeper') {
                shells = weapon.clipSize;
                document.getElementById('sweeper-shells-bar').style.width = '100%';
            } else {
                const needed = weapon.clipSize - ammo;
                const available = Math.min(needed, maxAmmo);
                ammo += available;
                maxAmmo -= available;
            }
            
            updateAmmoDisplay();
        }

        function updateAmmoDisplay() {
            const weapon = WEAPONS[currentWeapon];
            
            if (currentWeapon === 'sweeper') {
                ammoCount.textContent = shells;
                ammoType.textContent = weapon.ammoMode === 'buckshot' ? 'Buckshot' : 'Slug';
            } else {
                ammoCount.textContent = ammo;
                ammoType.textContent = weapon.ammoType;
            }
            
            // Update bars
            if (currentWeapon === 'anvil') {
                document.getElementById('anvil-heat-bar').style.width = `${(heat / weapon.maxHeat) * 100}%`;
            } else if (currentWeapon === 'judge') {
                document.getElementById('judge-ammo-bar').style.width = `${(ammo / weapon.clipSize) * 100}%`;
            }
        }

        // EFFECTS FUNCTIONS

        function playHeavyFireEffects() {
            // Muzzle flash
            const flash = document.createElement('div');
            flash.className = 'muzzleFlash';
            flash.style.left = `${player.x + Math.cos(player.angle) * 40 - 50}px`;
            flash.style.top = `${player.y + Math.sin(player.angle) * 40 - 50}px`;
            document.getElementById('gameContainer').appendChild(flash);
            
            // Animate flash
            let opacity = 1;
            const animate = () => {
                opacity -= 0.1;
                flash.style.opacity = opacity;
                if (opacity > 0) {
                    requestAnimationFrame(animate);
                } else {
                    flash.remove();
                }
            };
            animate();
            
            // Sound simulation (would use Web Audio API)
            playGunshotSound();
        }

        function applyScreenShake(intensity, duration) {
            cameraShake = intensity;
            
            const shakeInterval = setInterval(() => {
                screenShakeOffset.x = (Math.random() - 0.5) * cameraShake * 20;
                screenShakeOffset.y = (Math.random() - 0.5) * cameraShake * 20;
                cameraShake *= 0.9;
                
                if (cameraShake < 0.01) {
                    clearInterval(shakeInterval);
                    screenShakeOffset = { x: 0, y: 0 };
                }
            }, 16);
        }

        function playGunshotSound() {
            // Create audio context for each shot
            const audioContext = new (window.AudioContext || window.webkitAudioContext)();
            const oscillator = audioContext.createOscillator();
            const gainNode = audioContext.createGain();
            
            // Different sounds for different weapons
            switch(currentWeapon) {
                case 'anvil':
                    oscillator.frequency.setValueAtTime(80, audioContext.currentTime);
                    oscillator.type = 'sawtooth';
                    gainNode.gain.setValueAtTime(0.5, audioContext.currentTime);
                    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.3);
                    break;
                case 'judge':
                    oscillator.frequency.setValueAtTime(200, audioContext.currentTime);
                    oscillator.type = 'sine';
                    gainNode.gain.setValueAtTime(0.7, audioContext.currentTime);
                    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.2);
                    break;
                case 'sweeper':
                    oscillator.frequency.setValueAtTime(120, audioContext.currentTime);
                    oscillator.type = 'square';
                    gainNode.gain.setValueAtTime(1.0, audioContext.currentTime);
                    gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.5);
                    break;
            }
            
            oscillator.connect(gainNode);
            gainNode.connect(audioContext.destination);
            oscillator.start();
            oscillator.stop(audioContext.currentTime + 0.5);
        }

        function createRecoilIndicator() {
            const indicator = document.createElement('div');
            indicator.className = 'recoilIndicator';
            indicator.style.left = `${player.x}px`;
            indicator.style.top = `${player.y}px`;
            document.getElementById('gameContainer').appendChild(indicator);
            
            // Animate recoil indicator
            let size = 5;
            const animate = () => {
                size += 2;
                indicator.style.width = `${size}px`;
                indicator.style.height = `${size}px`;
                indicator.style.opacity -= 0.05;
                
                if (indicator.style.opacity > 0) {
                    requestAnimationFrame(animate);
                } else {
                    indicator.remove();
                }
            };
            animate();
        }

        // GAME LOOP

        function update() {
            // Move player based on input
            const weapon = WEAPONS[currentWeapon];
            let moveSpeed = PLAYER_SPEED * weapon.movementModifier;
            
            if (keys['w'] || keys['arrowup']) player.y -= moveSpeed;
            if (keys['s'] || keys['arrowdown']) player.y += moveSpeed;
            if (keys['a'] || keys['arrowleft']) player.x -= moveSpeed;
            if (keys['d'] || keys['arrowright']) player.x += moveSpeed;
            
            // Constrain to canvas
            player.x = Math.max(20, Math.min(CANVAS_WIDTH - 20, player.x));
            player.y = Math.max(20, Math.min(CANVAS_HEIGHT - 20, player.y));
            
            // Update player angle
            player.angle = Math.atan2(target.y - player.y, target.x - player.x);
            
            // Update bullets
            for (let i = bullets.length - 1; i >= 0; i--) {
                bullets[i].x += bullets[i].vx;
                bullets[i].y += bullets[i].vy;
                bullets[i].life--;
                
                // Remove old bullets
                if (bullets[i].life <= 0) {
                    bullets.splice(i, 1);
                }
            }
            
            // Animate recoil recovery
            recoil.x *= 0.9;
            recoil.y *= 0.9;
            
            // Cool down Anvil
            if (currentWeapon === 'anvil' && heat > 0 && !isOverheated) {
                heat = Math.max(0, heat - WEAPONS.anvil.cooldownRate / 60);
                heatFill.style.width = `${(heat / WEAPONS.anvil.maxHeat) * 100}%`;
            }
        }

        function draw() {
            // Clear with screen shake
            ctx.save();
            ctx.translate(screenShakeOffset.x, screenShakeOffset.y);
            ctx.clearRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);
            
            // Draw target
            ctx.fillStyle = '#ff4444';
            ctx.beginPath();
            ctx.arc(target.x, target.y, 15, 0, Math.PI * 2);
            ctx.fill();
            
            // Draw player
            ctx.save();
            ctx.translate(player.x, player.y);
            ctx.rotate(player.angle);
            
            // Player body
            ctx.fillStyle = '#4a9eff';
            ctx.fillRect(-15, -10, 30, 20);
            
            // Weapon indicator
            ctx.fillStyle = '#ff6b00';
            ctx.fillRect(15, -5, 25, 10);
            
            ctx.restore();
            
            // Draw bullets
            bullets.forEach(bullet => {
                ctx.fillStyle = bullet.type === 'sweeper' ? '#ffaa00' : 
                               bullet.type === 'anvil' ? '#ff6b00' : '#ffffff';
                ctx.beginPath();
                ctx.arc(bullet.x, bullet.y, 
                       bullet.type === 'anvil' ? 4 : 
                       bullet.type === 'sweeper' ? 3 : 2, 
                       0, Math.PI * 2);
                ctx.fill();
                
                // Trail effect
                ctx.beginPath();
                ctx.moveTo(bullet.x, bullet.y);
                ctx.lineTo(bullet.x - bullet.vx * 0.5, bullet.y - bullet.vy * 0.5);
                ctx.strokeStyle = ctx.fillStyle + '80';
                ctx.lineWidth = 2;
                ctx.stroke();
            });
            
            // Draw recoil indicator
            if (Math.abs(recoil.x) > 0.1 || Math.abs(recoil.y) > 0.1) {
                ctx.strokeStyle = '#ff000080';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(player.x, player.y);
                ctx.lineTo(player.x + recoil.x * 20, player.y + recoil.y * 20);
                ctx.stroke();
            }
            
            ctx.restore(); // Restore from screen shake
        }

        // FPS Counter
        let frameCount = 0;
        let lastTime = Date.now();
        
        function updateFPS() {
            frameCount++;
            const now = Date.now();
            if (now - lastTime >= 1000) {
                fpsCounter.textContent = `FPS: ${frameCount}`;
                frameCount = 0;
                lastTime = now;
            }
        }

        // Main game loop
        function gameLoop() {
            update();
            draw();
            updateFPS();
            requestAnimationFrame(gameLoop);
        }
