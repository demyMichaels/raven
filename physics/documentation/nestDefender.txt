    <script>
        // PHYSICS CONSTANTS
        const GRAVITY = 9.81; // m/s¬≤
        const DRAG_COEFFICIENT = 0.0001;
        const TIME_SCALE = 0.1; // Slow motion for visibility
        const PLAYER_MASS = 80; // kg
        const GROUND_FRICTION = 0.95;

        // WEAPON PHYSICS DATA (Realistic values)
        const WEAPONS = {
            anvil: {
                name: "THE ANVIL",
                mass: 13.0, // kg
                bulletMass: 0.00933, // kg (150 gr)
                bulletVelocity: 850, // m/s
                bulletRadius: 0.00762, // m (7.62mm)
                fireRate: 650, // RPM
                recoilImpulse: 35, // N‚ãÖs (calculated from momentum)
                heatPerShot: 25,
                maxHeat: 100,
                cooldownRate: 10, // per second
                bulletColor: '#ff4444',
                spreadAngle: 0.02, // radians
                maxAmmo: 100
            },
            judge: {
                name: "THE JUDGE",
                mass: 1.2, // kg
                bulletMass: 0.015, // kg (230 gr)
                bulletVelocity: 260, // m/s
                bulletRadius: 0.01143, // m (.45 cal)
                fireRate: 40, // RPM (semi-auto realistic)
                recoilImpulse: 8, // N‚ãÖs
                bulletColor: '#44aaff',
                spreadAngle: 0.01, // radians
                maxAmmo: 56,
                bulletDrop: 0.05 // Additional gravity effect for realism
            },
            sweeper: {
                name: "ROOM SWEEPER",
                mass: 3.8, // kg
                pelletMass: 0.0032, // kg per pellet (49 gr)
                bulletVelocity: 400, // m/s
                pelletRadius: 0.0085, // m (00 buck pellet)
                fireRate: 75, // RPM (pump action)
                recoilImpulse: 22, // N‚ãÖs
                bulletColor: '#ffaa00',
                spreadAngle: 0.08, // radians (wide spread)
                pelletsPerShot: 8,
                maxAmmo: 28
            }
        };

        // SIMULATION STATE
        let currentWeapon = 'anvil';
        let bullets = [];
        let player = {
            x: 400,
            y: 500,
            width: 60,
            height: 120,
            velocityX: 0,
            velocityY: 0,
            rotation: 0,
            isMoving: false,
            moveDirection: 0
        };
        
        let weaponState = {
            heat: 0,
            isOverheated: false,
            lastFireTime: 0,
            ammo: WEAPONS.anvil.maxAmmo,
            recoilRotation: 0,
            recoilOffsetY: 0
        };
        
        let isAutoFiring = false;
        let autoFireInterval = null;

        // CANVAS SETUP
        const canvas = document.getElementById('physicsCanvas');
        const ctx = canvas.getContext('2d');
        canvas.width = 800;
        canvas.height = 600;

        // PHYSICS CALCULATIONS
        function calculateRecoil() {
            const weapon = WEAPONS[currentWeapon];
            const totalMass = PLAYER_MASS + weapon.mass;
            
            // Conservation of momentum: m_bullet * v_bullet = m_total * v_recoil
            const recoilVelocity = weapon.recoilImpulse / totalMass;
            
            // Apply to player
            player.velocityX -= recoilVelocity * Math.cos(player.rotation);
            player.velocityY -= recoilVelocity * Math.sin(player.rotation);
            
            // Weapon climb (rotation from recoil)
            weaponState.recoilRotation += weapon.recoilImpulse / weapon.mass * 0.05;
            
            updatePhysicsDisplay(recoilVelocity);
            
            return recoilVelocity;
        }

        function calculateMovement() {
            const weapon = WEAPONS[currentWeapon];
            const totalMass = PLAYER_MASS + weapon.mass;
            
            // Movement speed affected by total mass
            const maxSpeed = 6 * (PLAYER_MASS / totalMass);
            
            if (player.isMoving) {
                player.velocityX += player.moveDirection * maxSpeed * 0.1;
            }
            
            // Apply friction
            player.velocityX *= GROUND_FRICTION;
            player.velocityY *= GROUND_FRICTION;
            
            // Apply gravity to player (if in air)
            if (player.y < 500) {
                player.velocityY += GRAVITY * 0.01;
            }
            
            // Update position
            player.x += player.velocityX;
            player.y += player.velocityY;
            
            // Keep player on ground
            if (player.y > 500) {
                player.y = 500;
                player.velocityY = 0;
            }
            
            // Keep in bounds
            player.x = Math.max(50, Math.min(canvas.width - 50, player.x));
            
            // Update mass display
            document.getElementById('playerMass').textContent = `${PLAYER_MASS} kg`;
            document.getElementById('totalMass').textContent = `${(PLAYER_MASS + weapon.mass).toFixed(1)} kg`;
            document.getElementById('moveSpeed').textContent = `${maxSpeed.toFixed(1)} m/s`;
        }

        function fireBullet() {
            if (weaponState.isOverheated || weaponState.ammo <= 0) return;
            
            const now = Date.now();
            const weapon = WEAPONS[currentWeapon];
            
            // Check fire rate
            if (now - weaponState.lastFireTime < 60000 / weapon.fireRate) return;
            
            weaponState.lastFireTime = now;
            weaponState.ammo--;
            
            // Calculate recoil
            const recoilVel = calculateRecoil();
            
            // Add heat for machine gun
            if (currentWeapon === 'anvil') {
                weaponState.heat += weapon.heatPerShot;
                if (weaponState.heat >= weapon.maxHeat) {
                    weaponState.isOverheated = true;
                    setTimeout(() => {
                        weaponState.isOverheated = false;
                        weaponState.heat = 0;
                    }, 3000);
                }
                document.getElementById('heatLevel').textContent = `${Math.min(100, weaponState.heat)}%`;
            }
            
            // Create bullet(s)
            if (currentWeapon === 'sweeper') {
                // Shotgun fires multiple pellets
                for (let i = 0; i < weapon.pelletsPerShot; i++) {
                    const spread = (Math.random() - 0.5) * weapon.spreadAngle;
                    createBulletInstance(player.rotation + spread, weapon, true);
                }
            } else {
                // Other weapons fire single bullets
                const spread = (Math.random() - 0.5) * weapon.spreadAngle;
                createBulletInstance(player.rotation + spread, weapon, false);
            }
            
            // Recoil animation
            weaponState.recoilOffsetY = 10;
            
            updateBulletDisplay();
        }

        function createBulletInstance(angle, weapon, isPellet) {
            const bullet = {
                x: player.x + Math.cos(angle) * 40,
                y: player.y + Math.sin(angle) * 40,
                velocityX: Math.cos(angle) * weapon.bulletVelocity * 0.1,
                velocityY: Math.sin(angle) * weapon.bulletVelocity * 0.1,
                mass: isPellet ? weapon.pelletMass : weapon.bulletMass,
                radius: weapon.bulletRadius * 50, // Scaled for visibility
                color: weapon.bulletColor,
                type: currentWeapon,
                life: 1000,
                gravityMultiplier: currentWeapon === 'judge' ? 1.5 : 0.2
            };
            
            bullets.push(bullet);
        }

        function updateBullets() {
            for (let i = bullets.length - 1; i >= 0; i--) {
                const bullet = bullets[i];
                
                // Apply gravity (more for Judge pistol)
                bullet.velocityY += GRAVITY * 0.001 * bullet.gravityMultiplier;
                
                // Apply drag
                bullet.velocityX *= (1 - DRAG_COEFFICIENT);
                bullet.velocityY *= (1 - DRAG_COEFFICIENT);
                
                // Update position
                bullet.x += bullet.velocityX * TIME_SCALE;
                bullet.y += bullet.velocityY * TIME_SCALE;
                
                // Reduce life
                bullet.life--;
                
                // Remove if out of bounds or dead
                if (bullet.x < -100 || bullet.x > canvas.width + 100 || 
                    bullet.y < -100 || bullet.y > canvas.height + 100 ||
                    bullet.life <= 0) {
                    bullets.splice(i, 1);
                }
            }
            
            document.getElementById('bulletCount').textContent = bullets.length;
        }

        function updatePhysicsDisplay(recoilVel) {
            document.getElementById('recoilVel').textContent = `${recoilVel.toFixed(2)} m/s`;
            document.getElementById('weaponClimb').textContent = `${(weaponState.recoilRotation * 180 / Math.PI).toFixed(1)}¬∞`;
        }

        function updateBulletDisplay() {
            const bulletsList = document.getElementById('bulletsList');
            bulletsList.innerHTML = '<h4>ACTIVE BULLETS IN FLIGHT:</h4>';
            
            // Group bullets by type
            const counts = { anvil: 0, judge: 0, sweeper: 0 };
            bullets.forEach(b => counts[b.type]++);
            
            for (const [type, count] of Object.entries(counts)) {
                if (count > 0) {
                    const div = document.createElement('div');
                    div.className = 'bullet-info';
                    div.innerHTML = `
                        <span>${WEAPONS[type].name}</span>
                        <span>${count} bullet${count !== 1 ? 's' : ''}</span>
                        <span>${WEAPONS[type].bulletVelocity} m/s</span>
                    `;
                    bulletsList.appendChild(div);
                }
            }
            
            if (bullets.length === 0) {
                const div = document.createElement('div');
                div.className = 'bullet-info';
                div.textContent = 'No bullets in flight';
                bulletsList.appendChild(div);
            }
        }

        // RENDERING
        function draw() {
            // Clear canvas
            ctx.fillStyle = '#0a0a1a';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Draw ground
            ctx.fillStyle = '#2a2a4a';
            ctx.fillRect(0, 500, canvas.width, 100);
            
            // Draw player
            ctx.save();
            ctx.translate(player.x, player.y);
            ctx.rotate(player.rotation + weaponState.recoilRotation);
            
            // Player body
            ctx.fillStyle = '#4a4a8a';
            ctx.fillRect(-player.width/2, -player.height/2, player.width, player.height);
            
            // Current weapon
            const weaponColor = WEAPONS[currentWeapon].bulletColor;
            ctx.fillStyle = weaponColor;
            ctx.fillRect(player.width/2, -5, 60, 10);
            
            ctx.restore();
            
            // Draw bullets
            bullets.forEach(bullet => {
                ctx.save();
                ctx.translate(bullet.x, bullet.y);
                
                // Bullet
                ctx.fillStyle = bullet.color;
                ctx.beginPath();
                ctx.arc(0, 0, bullet.radius, 0, Math.PI * 2);
                ctx.fill();
                
                // Velocity vector
                ctx.strokeStyle = bullet.color + '80';
                ctx.lineWidth = 2;
                ctx.beginPath();
                ctx.moveTo(0, 0);
                ctx.lineTo(bullet.velocityX * 2, bullet.velocityY * 2);
                ctx.stroke();
                
                ctx.restore();
            });
            
            // Draw stats overlay
            drawStats();
            
            // Recoil animation decay
            weaponState.recoilOffsetY *= 0.9;
            weaponState.recoilRotation *= 0.95;
        }

        function drawStats() {
            // Draw recoil vector
            ctx.strokeStyle = '#ff0000';
            ctx.lineWidth = 3;
            ctx.beginPath();
            ctx.moveTo(player.x, player.y);
            ctx.lineTo(
                player.x - player.velocityX * 20,
                player.y - player.velocityY * 20
            );
            ctx.stroke();
            
            // Draw velocity text
            const speed = Math.sqrt(player.velocityX**2 + player.velocityY**2);
            ctx.fillStyle = '#00ff88';
            ctx.font = '12px monospace';
            ctx.fillText(`Velocity: ${speed.toFixed(2)} m/s`, player.x + 70, player.y - 20);
        }

        // INPUT HANDLING
        document.querySelectorAll('.weapon').forEach(weapon => {
            weapon.addEventListener('click', () => {
                document.querySelectorAll('.weapon').forEach(w => w.classList.remove('active'));
                weapon.classList.add('active');
                currentWeapon = weapon.dataset.weapon;
                weaponState.ammo = WEAPONS[currentWeapon].maxAmmo;
                weaponState.heat = 0;
                weaponState.isOverheated = false;
                updateBulletDisplay();
            });
        });

        document.getElementById('fireBtn').addEventListener('click', fireBullet);
        
        document.getElementById('autoBtn').addEventListener('mousedown', () => {
            isAutoFiring = true;
            autoFireInterval = setInterval(fireBullet, 1000 / (WEAPONS[currentWeapon].fireRate / 60));
        });
        
        document.getElementById('autoBtn').addEventListener('mouseup', () => {
            isAutoFiring = false;
            clearInterval(autoFireInterval);
        });
        
        document.getElementById('autoBtn').addEventListener('mouseleave', () => {
            isAutoFiring = false;
            clearInterval(autoFireInterval);
        });

        document.getElementById('resetBtn').addEventListener('click', () => {
            bullets = [];
            player.x = 400;
            player.y = 500;
            player.velocityX = 0;
            player.velocityY = 0;
            weaponState.heat = 0;
            weaponState.isOverheated = false;
            weaponState.ammo = WEAPONS[currentWeapon].maxAmmo;
            updateBulletDisplay();
        });

        document.addEventListener('keydown', (e) => {
            if (e.code === 'Space') {
                fireBullet();
                e.preventDefault();
            }
            if (e.key === 'ArrowLeft' || e.key === 'a') {
                player.isMoving = true;
                player.moveDirection = -1;
            }
            if (e.key === 'ArrowRight' || e.key === 'd') {
                player.isMoving = true;
                player.moveDirection = 1;
            }
        });

        document.addEventListener('keyup', (e) => {
            if (e.key === 'ArrowLeft' || e.key === 'a' || 
                e.key === 'ArrowRight' || e.key === 'd') {
                player.isMoving = false;
            }
        });

        // Mouse aiming
        canvas.addEventListener('mousemove', (e) => {
            const rect = canvas.getBoundingClientRect();
            const mouseX = e.clientX - rect.left;
            const mouseY = e.clientY - rect.top;
            
            player.rotation = Math.atan2(mouseY - player.y, mouseX - player.x);
        });

        // GAME LOOP
        function gameLoop() {
            calculateMovement();
            updateBullets();
            draw();
            requestAnimationFrame(gameLoop);
        }

        // Initialize
        gameLoop();
        updateBulletDisplay();
    </script>

        // Update flow lines
        function updateFlowLines(deltaTime) {
            if (!showAirFlow) {
                flowLines.forEach(fl => fl.element.setAttribute('opacity', '0'));
                return;

            const speed = isSlowMo ? 0.3 : 1;
            flowLines.forEach((fl, i) => {
                fl.phase += deltaTime * 0.002 * speed;

                const baseY = 180 + Math.sin(fl.phase + i * 0.5) * 30;
                const waveY = Math.sin(fl.phase * 2) * 15;

                const startX = fl.x;
                const endX = fl.x + 50;
                const midX = (startX + endX) / 2;

                const d = `M${startX},${baseY} Q${midX},${baseY + waveY} ${endX},${baseY}`;
                fl.element.setAttribute('d', d);
                fl.element.setAttribute('opacity', (0.3 + Math.sin(fl.phase) * 0.2).toString());
            });
        }

        // Update particles
        function updateParticles(deltaTime) {
            const speed = isSlowMo ? 0.3 : 1;

            for (let i = particles.length - 1; i >= 0; i--) {
                const p = particles[i];
                p.x += p.vx * deltaTime * 0.05 * speed;
                p.y += p.vy * deltaTime * 0.05 * speed;
                p.life -= deltaTime * 0.002 * speed;

                p.element.setAttribute('cx', p.x);
                p.element.setAttribute('cy', p.y);
                p.element.setAttribute('opacity', p.life * 0.8);
                p.element.setAttribute('r', p.size * p.life);

                if (p.life <= 0) {
                    p.element.remove();
                    particles.splice(i, 1);
                }
            }
