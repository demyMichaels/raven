            setupDarkMode() {
                // Check for user's preferred color scheme
                if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
                    document.documentElement.classList.add('dark');
                }

                // Listen for changes in color scheme preference
                window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', event => {
                    if (event.matches) {
                        document.documentElement.classList.add('dark');
                    } else {
                        document.documentElement.classList.remove('dark');
                    }
                });

                // Manual dark mode toggle
                const darkModeToggle = document.getElementById('darkModeToggle');
                darkModeToggle.addEventListener('click', () => {
                    document.documentElement.classList.toggle('dark');
                    const icon = darkModeToggle.querySelector('.material-icons');
                    icon.textContent = document.documentElement.classList.contains('dark') ? 'light_mode' : 'dark_mode';
                });
            }

            setupEventListeners() {
                const tutorInput = document.getElementById('tutorInput');
                const sendButton = document.getElementById('tutorSend');

                sendButton.addEventListener('click', () => this.sendMessage());
                tutorInput.addEventListener('keypress', (e) => {
                    if (e.key === 'Enter' && !e.shiftKey) {
                        e.preventDefault();
                        this.sendMessage();
                    }
                });

                // Quick action buttons
                const quickActions = document.querySelectorAll('.quick-action');
                quickActions.forEach((button, index) => {
                    button.addEventListener('click', () => {
                        const prompts = [
                            "Can you explain a concept to me?",
                            "I need help with my homework",
                            "Can you give me some practice problems?",
                            "What are some effective study tips?"
                        ];
                        tutorInput.value = prompts[index];
                        tutorInput.focus();
                    });
                });
            }

            async sendMessage() {
                if (this.isProcessing) return;

                const input = document.getElementById('tutorInput');
                const sendButton = document.getElementById('tutorSend');
                const message = input.value.trim();

                if (!message) return;

                const chat = document.getElementById('tutorChat');

                // Add user message
                this.addChatMessage(chat, message, 'user');
                this.conversationHistory.push({ role: 'user', content: message });

                input.value = '';
                this.isProcessing = true;
                sendButton.disabled = true;

                // Add loading message with typing indicator
                const loadingDiv = this.addChatMessage(chat, '<div class="typing-indicator"><span></span><span></span><span></span></div>', 'bot', true);

                try {
                    const response = await this.getAIResponse(message);
                    this.conversationHistory.push({ role: 'assistant', content: response });
                    this.updateMessage(loadingDiv, response, 'bot');
                } catch (error) {
                    console.error('AI API error:', error);
                    this.updateMessage(loadingDiv, this.getErrorMessage(error), 'error');
                } finally {
                    this.isProcessing = false;
                    sendButton.disabled = false;
                    input.focus();
                }
            }

            async getAIResponse(message) {
                // Try DeepSeek first, then Gemini as fallback
                try {
                    return await this.tryDeepSeekAPI(message);
                } catch (deepseekError) {
                    console.warn('DeepSeek failed, trying Gemini:', deepseekError);
                    return await this.tryGeminiAPI(message);
                }
            }

            async tryDeepSeekAPI(message) {
                const messages = [
                    {
                        role: 'system',
                        content: this.getSystemPrompt()
                    },
                    ...this.conversationHistory.slice(-10) // Keep last 10 messages for context
                ];

                const response = await fetch(this.config.DEEPSEEK_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${this.config.DEEPSEEK_API_KEY}`
                    },
                    body: JSON.stringify({
                        model: 'deepseek-chat',
                        messages: messages,
                        max_tokens: this.config.MAX_TOKENS,
                        temperature: this.config.TEMPERATURE,
                        stream: false
                    })
                });

                if (!response.ok) {
                    const errorData = await response.json().catch(() => ({}));
                    throw new Error(`DeepSeek API error (${response.status}): ${errorData.error?.message || response.statusText}`);
                }

                const data = await response.json();
                return data.choices[0].message.content;
            }

            async tryGeminiAPI(message) {
                const url = `${this.config.GEMINI_URL}?key=${this.config.GEMINI_API_KEY}`;

                // Build conversation context for Gemini
                const conversationContext = this.conversationHistory
                    .slice(-8)
                    .map(msg => `${msg.role === 'user' ? 'User' : 'Assistant'}: ${msg.content}`)
                    .join('\n\n');

                const fullPrompt = `${this.getSystemPrompt()}\n\nConversation history:\n${conversationContext}`;

                const response = await fetch(url, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        contents: [{
                            parts: [{
                                text: fullPrompt
                            }]
                        }],
                        generationConfig: {
                            maxOutputTokens: this.config.MAX_TOKENS,
                            temperature: this.config.TEMPERATURE
                        },
                        safetySettings: [
                            {
                                category: "HARM_CATEGORY_HARASSMENT",
                                threshold: "BLOCK_MEDIUM_AND_ABOVE"
                            },
                            {
                                category: "HARM_CATEGORY_HATE_SPEECH",
                                threshold: "BLOCK_MEDIUM_AND_ABOVE"
                            },
                            {
                                category: "HARM_CATEGORY_SEXUALLY_EXPLICIT",
                                threshold: "BLOCK_MEDIUM_AND_ABOVE"
                            },
                            {
                                category: "HARM_CATEGORY_DANGEROUS_CONTENT",
                                threshold: "BLOCK_MEDIUM_AND_ABOVE"
                            }
                        ]
                    })
                });

                if (!response.ok) {
                    const errorData = await response.json().catch(() => ({}));
                    throw new Error(`Gemini API error (${response.status}): ${errorData.error?.message || response.statusText}`);
                }

                const data = await response.json();
                return data.candidates[0].content.parts[0].text;
            }

            formatMessage(message, type) {
                if (type === 'error') {
                    return `<span class="material-icons text-sm inline-block mr-1 align-middle">error</span>${message}`;
                }

                // Enhanced markdown-like formatting
                let formatted = message
                    // Headers
                    .replace(/### (.*?)(\n|$)/g, '<h3 class="text-lg font-semibold mt-3 mb-2">$1</h3>')
                    .replace(/## (.*?)(\n|$)/g, '<h2 class="text-xl font-semibold mt-3 mb-2">$1</h2>')
                    .replace(/# (.*?)(\n|$)/g, '<h1 class="text-2xl font-bold mt-3 mb-2">$1</h1>')
                    // Bold and italic
                    .replace(/\*\*\*(.*?)\*\*\*/g, '<strong><em>$1</em></strong>')
                    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                    .replace(/\*(.*?)\*/g, '<em>$1</em>')
                    // Code blocks
                    .replace(/```([\s\S]*?)```/g, '<pre class="bg-neutral-900 dark:bg-black text-neutral-100 p-3 rounded-lg overflow-x-auto my-2"><code>$1</code></pre>')
                    .replace(/`(.*?)`/g, '<code class="bg-neutral-200 dark:bg-neutral-700 px-1.5 py-0.5 rounded text-sm">$1</code>')
                    // Lists (unordered)
                    .replace(/^\s*[-*]\s+(.+)$/gm, '<li class="ml-4">$1</li>')
                    // Lists (ordered)
                    .replace(/^\s*\d+\.\s+(.+)$/gm, '<li class="ml-4">$1</li>')
                    // Line breaks
                    .replace(/\n\n/g, '</p><p class="mb-2">')
                    .replace(/\n/g, '<br>');

                // Wrap in paragraph if not already wrapped
                if (!formatted.includes('<h1') && !formatted.includes('<h2') && !formatted.includes('<h3')) {
                    formatted = '<p class="mb-2">' + formatted + '</p>';
                }

                // Wrap lists in ul/ol tags
                formatted = formatted.replace(/(<li class="ml-4">.*?<\/li>)+/g, match => {
                    return '<ul class="list-disc ml-6 my-2 space-y-1">' + match + '</ul>';
                });

                return formatted;
            }
